            // Close dropdown when clicking outside
            document.addEventListener('click', function (event) {
                if (window.innerWidth < 992) {
                    const dropdowns = document.querySelectorAll('.dropdown-menu');
                    dropdowns.forEach(menu => {
                        if (!menu.parentElement.contains(event.target)) {
                            menu.style.display = 'none';
                        }
                    });
                }
            });


            // Initialisation du menu déroulant pour "About Us" sur mobile
            const aboutLink = document.querySelector('.about-us-link');
            const dropdownIcon = document.querySelector('.mobile-dropdown-icon');
            const dropdownMenu = document.querySelector('.dropdown-menu');

            if (aboutLink) {
                aboutLink.addEventListener('click', function (e) {
                    if (window.innerWidth < 992) {
                        e.preventDefault();
                        const isOpen = dropdownMenu.style.display === 'block';

                        // Fermer tous les autres menus déroulants
                        document.querySelectorAll('.dropdown-menu').forEach(menu => {
                            if (menu !== dropdownMenu) {
                                menu.style.display = 'none';
                            }
                        });

                        // Basculer le menu actuel
                        dropdownMenu.style.display = isOpen ? 'none' : 'block';

                        // Rotation de l'icône
                        dropdownIcon.classList.toggle('rotate', !isOpen);
                    }
                });
            }
